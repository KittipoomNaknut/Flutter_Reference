main() + runApp()

จุดเริ่มต้นของทุก Flutter app — เหมือนกัน 100% ไม่ว่า app จะเล็กหรือใหญ่แค่ไหน

โครงสร้างพื้นฐาน ▼

main() คือ entry point ของโปรแกรม Dart — Flutter วิ่งมาที่นี่ก่อนเสมอ runApp() รับ Widget แรกไปแสดงบนหน้าจอ

💡 ไม่ต้องแก้ส่วนนี้เลย — copy ไปใช้ได้ทุก project แก้แค่ชื่อ title และ home:
import 'package:flutter/material.dart';

// จุดเริ่มต้นทุก Flutter app — เหมือนกันทุกครั้ง
void main() {
  runApp(const MyApp());
}

// MyApp — ตั้งค่า app ครั้งเดียว
class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'ชื่อ App',        // ← เปลี่ยนตรงนี้
      home: const MyHomePage(), // ← และตรงนี้
    );
  }
}
ส่วนหน้าที่
main()Entry point ของ Dartตายตัว
runApp()ส่ง Widget แรกไปแสดงตายตัว
MyAppWidget root — ตั้งค่า appชื่อเปลี่ยนได้
App-wide

MaterialApp

ตั้งค่าระดับ app ทั้งหมด — theme, navigation, locale — แก้ที่เดียว ใช้ทั้ง app

Properties ที่ใช้บ่อย ▼
PropertyTypeหน้าที่
titleStringชื่อ app ใน task switcher
homeWidgetหน้าแรก สำคัญ
themeThemeDataสี, font ทั้ง app optional
darkThemeThemeDatatheme dark mode optional
routesMap<String,Widget>แผนที่หน้าจอทั้งหมด optional
debugShowCheckedModeBannerboolซ่อน DEBUG banner optional
ตัวอย่าง: แบบ minimal ▼
MaterialApp(
  title: 'My App',
  home: const HomeScreen(),
  debugShowCheckedModeBanner: false, // ซ่อน DEBUG
)
ตัวอย่าง: เพิ่ม Theme (สีทั้ง app) ▼

กำหนดสีหลักครั้งเดียว — ทุก Widget ใช้สีนี้อัตโนมัติ

MaterialApp(
  title: 'My App',
  theme: ThemeData(
    colorScheme: ColorScheme.fromSeed(
      seedColor: Colors.blue,
    ),
    useMaterial3: true,
  ),
  home: const HomeScreen(),
)
ตัวอย่าง: เพิ่ม Routes (หลายหน้า) ▼
📌 เรียนลึกขึ้นใน Week 4 — ดูก่อนเพื่อรู้ว่ามีอยู่
MaterialApp(
  routes: {
    '/':         (ctx) => const HomeScreen(),
    '/profile':  (ctx) => const ProfileScreen(),
    '/settings': (ctx) => const SettingsScreen(),
  },
)

// วิธี navigate ไปหน้าอื่น
Navigator.pushNamed(context, '/profile');

Column & Row

จัดเรียง children แนวตั้ง (Column) หรือแนวนอน (Row) — ใช้บ่อยที่สุดใน Flutter

Visual: Column vs Row ▼
Column (↓ แนวตั้ง)
Widget A
Widget B
Widget C
Row (→ แนวนอน)
A
B
C
mainAxisAlignment — จัด children ในแนวหลัก ▼

Column: แนวหลัก = แนวตั้ง | Row: แนวหลัก = แนวนอน

ค่าผลลัพธ์
startชิดต้น (default) — ชิดบน/ซ้าย
centerกึ่งกลาง
endชิดปลาย — ชิดล่าง/ขวา
spaceBetweenกระจาย — ปลายสองข้างไม่มีช่องว่าง
spaceAroundกระจาย — มีช่องว่างรอบทุก widget
spaceEvenlyกระจายเท่ากันทุกช่อง
Column(
  mainAxisAlignment: MainAxisAlignment.center,
  crossAxisAlignment: CrossAxisAlignment.start,
  children: [
    Text('บรรทัดที่ 1'),
    Text('บรรทัดที่ 2'),
  ],
)
ตัวอย่าง: ซ้อน Column ใน Row (layout จริง) ▼
Row(
  children: [
    const CircleAvatar(radius: 24),
    const SizedBox(width: 12),
    // ซ้อน Column ข้างใน Row
    Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Text('Alice',
          style: TextStyle(fontWeight: FontWeight.bold)),
        Text('นักศึกษา',
          style: TextStyle(color: Colors.grey)),
      ],
    ),
  ],
)

Expanded & Flexible

แบ่งพื้นที่ใน Row/Column — ปุ่ม 0 ในเครื่องคิดเลขที่กว้าง 2 ช่องใช้ตัวนี้

Expanded vs Flexible + ตัวอย่าง flex ▼
ExpandedFlexible
พื้นที่ขอทั้งหมดที่เหลือขอแค่ที่ต้องการ
ใช้เมื่ออยากให้เต็มพื้นที่อยากให้ยืดหยุ่น
Row(
  children: [
    Expanded(
      flex: 2, // ขอ 2 ส่วน จากทั้งหมด 3 ส่วน
      child: Container(color: Colors.blue),
    ),
    Expanded(
      flex: 1, // ขอ 1 ส่วน
      child: Container(color: Colors.red),
    ),
  ],
)

// จากเครื่องคิดเลข — ปุ่ม 0 กว้าง 2 ช่อง
Row(
  children: [
    CalcButton(label: '0', flex: 2),
    CalcButton(label: '.', flex: 1),
    CalcButton(label: '=', flex: 1),

SizedBox & Padding

เพิ่มช่องว่างและระยะห่าง — ใช้แทน margin/padding ใน Container เมื่อไม่ต้องการ decoration

SizedBox vs Padding — ใช้อันไหน? ▼
💡 SizedBox = ช่องว่างระหว่าง Widget | Padding = ระยะห่างรอบ Widget เดียว
// SizedBox — ช่องว่างระหว่าง Widget
Column(children: [
  Text('ข้อความแรก'),
  const SizedBox(height: 16), // spacer แนวตั้ง
  Text('ข้อความที่สอง'),
])

Row(children: [
  Icon(Icons.star),
  const SizedBox(width: 8), // spacer แนวนอน
  Text('ดาวดวงโปรด'),
])

// Padding — เพิ่ม padding รอบ Widget เดียว
Padding(
  padding: const EdgeInsets.symmetric(
    horizontal: 16, vertical: 8,
  ),
  child: Text('ข้อความมี padding'),
)

Text

TextStyle + RichText ▼
Text(

    fontWeight: FontWeight.bold,

    height: 1.5, // line height

  textAlign: TextAlign.center,
  maxLines: 2,

Text.rich(

Image

3 แบบของ Image ▼

Image.network(

  width: 200, height: 200,

// 3. CircleAvatar — รูปวงกลม

💡 BoxFit.cover = เต็มกรอบ ตัดขอบ | BoxFit.contain = เต็มกรอบ ไม่ตัด | BoxFit.fill = ยืดเต็ม

ListView

รายการที่ scroll ได้ — ใช้แทน Column เมื่อ item มาก หรือไม่รู้จำนวนล่วงหน้า

3 แบบของ ListView ▼
// 1. ListView — item น้อย fixed
ListView(
  children: const [
    ListTile(title: Text('Item 1')),
    ListTile(title: Text('Item 2')),
  ],
)

// 2. ListView.builder — item มาก / ไม่รู้จำนวน
// สร้าง Widget เฉพาะที่แสดงบนหน้าจอ = ประหยัด memory
ListView.builder(
  itemCount: students.length,
  itemBuilder: (context, index) {
    return ListTile(
      title: Text(students[index].name),
      subtitle: Text('คะแนน: ${students[index].score}'),
      leading: const CircleAvatar(
        child: Icon(Icons.person)),
    );
  },
)

// 3. ListView.separated — มีเส้นคั่น
ListView.separated(
  itemCount: items.length,
  itemBuilder: (ctx, i) => Text(items[i]),
  separatorBuilder: (ctx, i) => const Divider(),
)
💡 ใช้ Column เมื่อ item น้อย (<10) — ใช้ ListView.builder เมื่อ item มาก เพราะ lazy load

Buttons — 3 แบบ

Logic เหมือนกันทุกอย่าง — ต่างกันแค่รูปลักษณ์ เลือกตามลำดับความสำคัญของ action

Visual Preview + เลือกใช้อันไหน ▼
ปุ่มทั้ง 3 แบบ
บันทึกดูเพิ่มเติมยกเลิก
Buttonใช้เมื่อตัวอย่าง
ElevatedButtonPrimary — สำคัญที่สุดบันทึก, Login, ซื้อ
OutlinedButtonAlternative — ทางเลือกดูเพิ่มเติม, แก้ไข
TextButtonSecondary — เบาที่สุดยกเลิก, ข้าม
Code ทั้ง 3 แบบ + Style + Disabled ▼
// ElevatedButton — มี background สี
ElevatedButton(
  onPressed: () {},
  style: ElevatedButton.styleFrom(
    backgroundColor: Colors.blue,
    foregroundColor: Colors.white,
    shape: RoundedRectangleBorder(
      borderRadius: BorderRadius.circular(8)),
  ),
  child: const Text('บันทึก'),
)

// OutlinedButton — มีแค่กรอบ
OutlinedButton(
  onPressed: () {},
  style: OutlinedButton.styleFrom(
    side: const BorderSide(color: Colors.blue),
  ),
  child: const Text('ดูเพิ่มเติม'),
)

// TextButton — ไม่มี background
TextButton(
  onPressed: () {},
  child: const Text('ยกเลิก'),

// Disabled — ส่ง null ให้ onPressed
ElevatedButton(

// Disabled style กำหนดเองได้
ElevatedButton(

TextField

รับ input ข้อความ — ใช้ร่วมกับ TextEditingController เพื่อดึงค่า

TextField พร้อม Controller ▼
class _MyState extends State<MyWidget> {
  final _controller = TextEditingController();

  @override
  void dispose() {
    _controller.dispose(); // ต้อง dispose เสมอ!
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Column(children: [
      TextField(
        controller: _controller,
        decoration: const InputDecoration(
          labelText: 'ชื่อ',
          hintText: 'พิมพ์ชื่อของคุณ',
          border: OutlineInputBorder(),
          prefixIcon: Icon(Icons.person),
        ),
        onChanged: (v) { /* เรียกทุกครั้งที่พิมพ์ */ },
        onSubmitted: (v) { /* กด Enter */ },
      ),
      ElevatedButton(
        onPressed: () {
          final text = _controller.text; // ดึงค่า
          print(text);
        },
        child: const Text('บันทึก'),
      ),
    ]);
  }
}

StatelessWidget

Widget ที่ UI ขึ้นอยู่กับ input เท่านั้น — ไม่มีข้อมูลภายในที่เปลี่ยนได้

โครงสร้างและตัวอย่าง ▼
💡 ใช้เมื่อ Widget ไม่ต้องเปลี่ยนตัวเอง — แค่รับข้อมูลมาแสดง
class StudentCard extends StatelessWidget {
  final String name;
  final int score;

  const StudentCard({
    super.key,
    required this.name,
    required this.score,
  });

  @override
  Widget build(BuildContext context) {
    return Card(
      child: ListTile(
        title: Text(name),
        trailing: Text('$score'),
      ),
    );
  }
}

// ใช้งาน
StudentCard(name: 'Alice', score: 85)
State

StatefulWidget

Widget ที่มีข้อมูลภายในเปลี่ยนแปลงได้ — แยกเป็น 2 class เสมอ

ทำไมต้อง 2 class? ▼
Widget (immutable) + State (mutable) = StatefulWidget

Widget ถูกสร้างและทิ้งบ่อยมากตอน rebuild แต่ State ต้องอยู่รอดได้ข้าม rebuild — ถ้าเก็บข้อมูลใน Widget มันจะหายทุกครั้ง

class Counter extends StatefulWidget {
  const Counter({super.key});

  @override
  State<Counter> createState() => _CounterState();
}

class _CounterState extends State<Counter> {
  // State อยู่ที่นี่ — อยู่รอดข้าม rebuild
  int _count = 0;

  @override
  Widget build(BuildContext context) {
    return Text('$_count');
  }
}
State

setState

บอก Flutter ว่า state เปลี่ยนแล้ว — ให้ build() ใหม่

setState ทำงานยังไง? ▼
1. ผู้ใช้กดปุ่ม
↓
2. setState(() { _count++; })
↓
3. Flutter mark widget ว่า "dirty"
↓
4. build() ถูกเรียกใหม่
↓
5. UI อัปเดตเฉพาะส่วนที่เปลี่ยน
// ✅ ถูก — แก้ state ใน setState
setState(() {
  _count++;
  _name = 'Alice';
});

// ❌ ผิด — แก้ state นอก setState
// UI ไม่อัปเดต!
_count++; // ค่าเปลี่ยนในหน่วยความจำ แต่จอไม่รู้
State

Lifecycle

ลำดับการทำงานของ StatefulWidget ตั้งแต่เกิดจนตาย

Methods ที่ต้องรู้ ▼
class _MyState extends State<MyWidget> {

  // 1. initState — ทำครั้งเดียวตอนสร้าง
  // ใช้: โหลดข้อมูลครั้งแรก, ตั้งค่า controller
  @override
  void initState() {
    super.initState(); // ต้องเรียกก่อนเสมอ
    loadData();
  }

  // 2. build — ทำทุกครั้งที่ setState ถูกเรียก
  @override
  Widget build(BuildContext context) {
    return Text('Hello');
  }

  // 3. dispose — ตอน Widget ถูกลบออกจาก tree
  // ใช้: ยกเลิก timer, dispose controller
  @override
  void dispose() {
    _controller.dispose();
    super.dispose(); // ต้องเรียกทีหลังเสมอ
  }
}

ตัวอย่าง: CalcButton จากเครื่องคิดเลข ▼
class CalcButton extends StatelessWidget {
  final String label;
  final VoidCallback? onTap;
  final Color color;
  final int flex;

  const CalcButton({
    super.key,
    required this.label,
    required this.onTap,
    this.color = Colors.grey,
    this.flex = 1,
  });

  @override
  Widget build(BuildContext context) {
    return Expanded(
      flex: flex,
      child: ElevatedButton(
        style: ElevatedButton.styleFrom(
          backgroundColor: color),
        onPressed: onTap,
        child: Text(label,
          style: const TextStyle(fontSize: 24)),
      ),
    );
  }
}

// ใช้งาน — สะอาด อ่านเข้าใจทันที
CalcButton(label: '7', onTap: () => _onNumber('7'))
CalcButton(label: '0', flex: 2, onTap: () => _onNumber('0'))

ส่ง Parameter ให้ Widget

วิธีส่งข้อมูลและ callback จาก Parent → Child Widget

Data + Callback Pattern ▼
💡 หลักการ: Child ไม่จัดการ state เอง — ส่ง event กลับ Parent แล้วให้ Parent setState
// Child รับข้อมูล และส่ง event กลับ
class LikeButton extends StatelessWidget {
  final bool isLiked;           // ข้อมูลจาก Parent
  final VoidCallback onToggle; // callback → Parent

  const LikeButton({
    super.key,
    required this.isLiked,
    required this.onToggle,
  });

  @override
  Widget build(BuildContext context) {
    return IconButton(
      icon: Icon(
        isLiked ? Icons.favorite : Icons.favorite_border,
        color: isLiked ? Colors.red : null,
      ),
      onPressed: onToggle,
    );
  }
}

// Parent จัดการ state
class _ParentState extends State<Parent> {
  bool _liked = false;

  @override
  Widget build(BuildContext context) {
    return LikeButton(
      isLiked: _liked,
      onToggle: () => setState(() => _liked = !_liked),
    );
  }
}
